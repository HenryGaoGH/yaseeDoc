@override
Future<List<CheckEntity>?> checks(String mac) async {
  final maps = await methodChannel.invokeListMethod("checks",mac);
  return maps?.map((e) => CheckEntity.fromJson(e.cast<String,dynamic>())).toList() ?? [];
}
